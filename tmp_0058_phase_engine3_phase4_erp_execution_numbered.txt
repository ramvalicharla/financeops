  146: CREATE TABLE IF NOT EXISTS e3_erp_posting_runs (
  147:     id TEXT PRIMARY KEY,
  148:     tenant_id TEXT NOT NULL,
  149:     workspace_id TEXT NOT NULL,
  150:     erp_connection_id TEXT NOT NULL,
  151:     je_draft_id TEXT NOT NULL,
  152:     idempotency_key TEXT NOT NULL,
  153:     execution_status TEXT NOT NULL CHECK(execution_status IN (
  154:         'QUEUED',
  155:         'SENT',
  156:         'PENDING',
  157:         'POSTED',
  158:         'FAILED_SOFT',
  159:         'FAILED_HARD',
  160:         'REVERSED',
  161:         'CANCELLED'
  162:     )),
  163:     workflow_state_snapshot TEXT NOT NULL,
  164:     narrative_hash_snapshot TEXT NOT NULL CHECK(length(narrative_hash_snapshot) = 64),
  165:     evidence_hash_snapshot TEXT NOT NULL CHECK(length(evidence_hash_snapshot) = 64),
  166:     approval_decision_id TEXT,
  167:     execution_snapshot_json TEXT NOT NULL DEFAULT '{}' CHECK(json_valid(execution_snapshot_json)),
  168:     external_document_id TEXT,
  169:     error_code TEXT,
  170:     error_message TEXT,
  171:     reversal_of_posting_run_id TEXT,
  172:     created_at TEXT NOT NULL,
  173:     completed_at TEXT,
  174:     FOREIGN KEY(erp_connection_id) REFERENCES e3_erp_connections(id) ON DELETE RESTRICT,
  175:     FOREIGN KEY(je_draft_id) REFERENCES e3_je_drafts(id) ON DELETE RESTRICT,
  176:     FOREIGN KEY(approval_decision_id) REFERENCES e3_je_decisions(id) ON DELETE RESTRICT,
  177:     FOREIGN KEY(reversal_of_posting_run_id) REFERENCES e3_erp_posting_runs(id) ON DELETE RESTRICT,
  178:     FOREIGN KEY(tenant_id) REFERENCES iam_tenants(tenant_id),
  179:     FOREIGN KEY(workspace_id) REFERENCES iam_workspaces(workspace_id)
  180: );

  183: ON e3_erp_posting_runs(tenant_id, workspace_id, je_draft_id, created_at DESC);
  184: 
  185: CREATE INDEX IF NOT EXISTS idx_e3_erp_posting_runs_scope_status
  186: ON e3_erp_posting_runs(tenant_id, workspace_id, execution_status, created_at DESC);
  187: 
  188: CREATE UNIQUE INDEX IF NOT EXISTS idx_e3_erp_posting_runs_lock
  189: ON e3_erp_posting_runs(tenant_id, workspace_id, je_draft_id, execution_status)
  190: WHERE execution_status IN ('QUEUED', 'SENT');
  191: 
  192: CREATE UNIQUE INDEX IF NOT EXISTS idx_e3_erp_posting_runs_idempotency
  193: ON e3_erp_posting_runs(tenant_id, workspace_id, idempotency_key);
  194: 
  195: CREATE TRIGGER IF NOT EXISTS trg_e3_erp_posting_runs_update_guard
  196: BEFORE UPDATE ON e3_erp_posting_runs
  197: FOR EACH ROW
  198: BEGIN
  199:     SELECT
  200:         CASE
  201:             WHEN
  202:                 OLD.id = NEW.id
  203:                 AND OLD.tenant_id = NEW.tenant_id
  204:                 AND OLD.workspace_id = NEW.workspace_id
  205:                 AND OLD.erp_connection_id = NEW.erp_connection_id
  206:                 AND OLD.je_draft_id = NEW.je_draft_id
  207:                 AND OLD.idempotency_key = NEW.idempotency_key
  208:                 AND OLD.workflow_state_snapshot = NEW.workflow_state_snapshot
  209:                 AND OLD.narrative_hash_snapshot = NEW.narrative_hash_snapshot
  210:                 AND OLD.evidence_hash_snapshot = NEW.evidence_hash_snapshot
  211:                 AND ifnull(OLD.approval_decision_id, '') = ifnull(NEW.approval_decision_id, '')
  212:                 AND OLD.execution_snapshot_json = NEW.execution_snapshot_json
  213:                 AND ifnull(OLD.reversal_of_posting_run_id, '') = ifnull(NEW.reversal_of_posting_run_id, '')
  214:                 AND OLD.created_at = NEW.created_at
  215:             THEN 1
  216:             ELSE RAISE(ABORT, 'e3_erp_posting_runs immutable snapshot fields cannot be updated')
  217:         END;
  218: 
  219:     SELECT
  220:         CASE
  221:             WHEN OLD.execution_status = NEW.execution_status THEN 1
  222:             WHEN OLD.execution_status = 'QUEUED' AND NEW.execution_status IN ('SENT', 'FAILED_SOFT', 'FAILED_HARD', 'CANCELLED') THEN 1
  223:             WHEN OLD.execution_status = 'SENT' AND NEW.execution_status IN ('PENDING', 'POSTED', 'FAILED_SOFT', 'FAILED_HARD') THEN 1
  224:             WHEN OLD.execution_status = 'PENDING' AND NEW.execution_status IN ('POSTED', 'FAILED_SOFT', 'FAILED_HARD') THEN 1
  225:             WHEN OLD.execution_status = 'FAILED_SOFT' AND NEW.execution_status IN ('SENT', 'CANCELLED') THEN 1
  226:             WHEN OLD.execution_status = 'PENDING' AND NEW.execution_status IN ('SENT') THEN 1
  227:             WHEN OLD.execution_status = 'POSTED' AND NEW.execution_status IN ('REVERSED') THEN 1
  228:             ELSE RAISE(ABORT, 'e3_erp_posting_runs status transition not allowed')
  229:         END;
  230: END;
  231: 
  232: CREATE TRIGGER IF NOT EXISTS trg_e3_erp_posting_runs_append_only_delete
  233: BEFORE DELETE ON e3_erp_posting_runs
  234: FOR EACH ROW
  235: BEGIN
  236:     SELECT RAISE(ABORT, 'e3_erp_posting_runs is append-only for delete');
  237: END;
