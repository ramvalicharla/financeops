   31: def _run_write_with_retry(fn):
   32:     attempts = int(getattr(_db, "DB_RETRY_ATTEMPTS", 5) or 5)
   33:     delay_seconds = float(getattr(_db, "DB_RETRY_DELAY_SECONDS", 0.2) or 0.2)
   34:     for attempt in range(attempts):
   35:         try:
   36:             return fn()
   37:         except sqlite3.OperationalError as exc:
   38:             if not _is_sqlite_lock_error(exc) or attempt >= attempts - 1:
   39:                 raise
   40:             time.sleep(delay_seconds)

   88:     refresh_token = str(token_payload.get("refresh_token") or "").strip()

  116: def set_connection_secret_payload(
  117:     *,
  118:     tenant_id: str,
  119:     workspace_id: str,
  120:     secret_ref: str,
  121:     payload: dict[str, Any],
  122:     conn: Any | None = None,
  123: ) -> dict[str, Any]:
  124:     _require_schema()
  125:     tenant = _db._normalize_required_text(tenant_id, field_name="tenant_id")
  126:     workspace = _db._normalize_required_text(workspace_id, field_name="workspace_id")
  127:     ref = _db._normalize_required_text(secret_ref, field_name="secret_ref")
  128:     encrypted, kid = _encrypt_secret_payload(payload)
  129:     now = _db.utc_now_iso()
  130:     def _upsert(db_conn: Any) -> None:
  131:         existing = db_conn.execute(
  132:             """
  133:             SELECT secret_ref
  134:             FROM e3_erp_secret_blobs
  135:             WHERE tenant_id = ?
  136:               AND workspace_id = ?
  137:               AND secret_ref = ?
  138:             LIMIT 1
  139:             """,
  140:             (tenant, workspace, ref),
  141:         ).fetchone()
  142:         if existing:
  143:             db_conn.execute(
  144:                 """
  145:                 UPDATE e3_erp_secret_blobs
  146:                 SET secret_ciphertext = ?,
  147:                     secret_kid = ?,
  148:                     updated_at = ?
  149:                 WHERE tenant_id = ?
  150:                   AND workspace_id = ?
  151:                   AND secret_ref = ?
  152:                 """,
  153:                 (encrypted, kid, now, tenant, workspace, ref),
  154:             )
  155:             return
  156:         db_conn.execute(
  157:             """
  158:             INSERT INTO e3_erp_secret_blobs (
  159:                 secret_ref, tenant_id, workspace_id, secret_ciphertext, secret_kid, created_at, updated_at
  160:             ) VALUES (?, ?, ?, ?, ?, ?, ?)
  161:             """,
  162:             (ref, tenant, workspace, encrypted, kid, now, now),
  163:         )
  164: 
  165:     if conn is not None:
  166:         _upsert(conn)
  167:     else:
  168:         def _write_secret() -> None:
  169:             with _db.get_conn() as write_conn:
  170:                 _upsert(write_conn)
  171: 
  172:         _run_write_with_retry(_write_secret)
  173:     return {"secret_ref": ref, "updated_at": now}

  208:     *,

  268:         state_token=state_token,
  269:         pkce_verifier=pkce_verifier,
  270:     )

  272:         "oauth_session_id": session_id,
  273:         "state_token": state_token,

  279: def consume_oauth_callback(
  280:     *,
  281:     tenant_id: str,
  282:     workspace_id: str,
  283:     connection_id: str,
  284:     state_token: str,
  285:     code: str,
  286:     realm_id: str | None = None,
  287: ) -> dict[str, Any]:
  288:     _require_schema()
  289:     tenant, workspace, connection = _normalize_scope(
  290:         tenant_id=tenant_id,
  291:         workspace_id=workspace_id,
  292:         connection_id=connection_id,
  293:     )
  294:     token = _db._normalize_required_text(state_token, field_name="state_token")
  295:     normalized_code = _db._normalize_required_text(code, field_name="code")
  296: 
  297:     conn_row = erp_repository.get_connection(
  298:         tenant_id=tenant,
  299:         workspace_id=workspace,
  300:         connection_id=connection,
  301:     )
  302:     if conn_row is None:
  303:         raise LookupError("erp_connection_id was not found.")
  304:     provider = _normalize_provider(str(conn_row.get("erp_type") or ""))
  305:     adapter = _adapter_for(provider)
  306:     discovered_org_id: str | None = None
  307:     discovered_realm_id: str | None = str(realm_id or "").strip() or None
  308: 
  309:     with _db.get_conn() as conn:
  310:         row = conn.execute(
  311:             """
  312:             SELECT *
  313:             FROM e3_erp_oauth_sessions
  314:             WHERE tenant_id = ?
  315:               AND workspace_id = ?
  316:               AND erp_connection_id = ?
  317:               AND state_token = ?
  318:             ORDER BY created_at DESC, id DESC
  319:             LIMIT 1
  320:             """,
  321:             (tenant, workspace, connection, token),
  322:         ).fetchone()
  323:         if not row:
  324:             raise RuntimeError("ERP_OAUTH_SESSION_NOT_FOUND")
  325:         session = dict(row)
  326:         status = _normalize_session_status(session.get("session_status"))
  327:         if status != "INITIATED":
  328:             raise RuntimeError("ERP_OAUTH_SESSION_REPLAY_BLOCKED")
  329:         expires_at = str(session.get("expires_at") or "")
  330:         if expires_at and datetime.fromisoformat(expires_at.replace("Z", "+00:00")) <= datetime.now(timezone.utc):
  331:             conn.execute(
  332:                 "UPDATE e3_erp_oauth_sessions SET session_status = 'EXPIRED' WHERE id = ?",
  333:                 (str(session["id"]),),
  334:             )
  335:             raise RuntimeError("ERP_OAUTH_SESSION_EXPIRED")
  336: 
  337:         try:
  338:             token_payload = adapter.exchange_oauth_code(
  339:                 connection_config=conn_row.get("connection_config") or {},
  340:                 redirect_uri=str(session.get("redirect_uri") or ""),
  341:                 code=normalized_code,
  342:                 pkce_verifier=str(session.get("pkce_verifier") or "") or None,
  343:             )
  344:             if provider == "QUICKBOOKS_ONLINE":
  345:                 token_payload = _db._normalize_engine3_json_object(
  346:                     token_payload,
  347:                     field_name="token_payload",
  348:                 )
  349:                 discovered_realm_id = discovered_realm_id or str(token_payload.get("realm_id") or "").strip() or None
  350:                 if discovered_realm_id:
  351:                     token_payload["realm_id"] = discovered_realm_id
  352:             normalized_token_payload = _db._normalize_engine3_json_object(
  353:                 token_payload,
  354:                 field_name="token_payload",
  355:             )
  356:             set_connection_secret_payload(
  357:                 tenant_id=tenant,
  358:                 workspace_id=workspace,
  359:                 secret_ref=str(conn_row.get("encrypted_credentials_ref") or ""),
  360:                 payload={
  361:                     "provider": provider,
  362:                     "token_payload": normalized_token_payload,
  363:                     "updated_at": _db.utc_now_iso(),
  364:                 },
  365:                 conn=conn,
  366:             )
  367: 
  368:             if provider == "ZOHO_BOOKS":
  369:                 organizations_response = adapter.get_organizations(
  370:                     connection_config=conn_row.get("connection_config") or {},
  371:                     credentials=normalized_token_payload,
  372:                 )
  373:                 if str(organizations_response.get("status") or "").upper() != "SUCCESS":
  374:                     error_message = str(
  375:                         organizations_response.get("error_message")
  376:                         or organizations_response.get("error_code")
  377:                         or "organization discovery failed"
  378:                     ).strip()
  379:                     raise RuntimeError(f"ERP_ZOHO_ORG_DISCOVERY_FAILED:{error_message}")
  380:                 organizations = organizations_response.get("organizations")
  381:                 first_organization = organizations[0] if isinstance(organizations, list) and organizations else {}
  382:                 discovered_org_id = _db._normalize_required_text(
  383:                     (first_organization or {}).get("organization_id"),
  384:                     field_name="organization_id",
  385:                 )
  386:                 updated_connection_config = _db._normalize_engine3_json_object(
  387:                     conn_row.get("connection_config") or {},
  388:                     field_name="connection_config",
  389:                 )
  390:                 updated_connection_config["organization_id"] = discovered_org_id
  391:                 conn.execute(
  392:                     """
  393:                     UPDATE e3_erp_connections
  394:                     SET connection_config_json = ?
  395:                     WHERE tenant_id = ?
  396:                       AND workspace_id = ?
  397:                       AND id = ?
  398:                     """,
  399:                     (
  400:                         _db._canonical_json_text(updated_connection_config),
  401:                         tenant,
  402:                         workspace,
  403:                         connection,
  404:                     ),
  405:                 )
  406:             if provider == "QUICKBOOKS_ONLINE" and discovered_realm_id:
  407:                 updated_connection_config = _db._normalize_engine3_json_object(
  408:                     conn_row.get("connection_config") or {},
  409:                     field_name="connection_config",
  410:                 )
  411:                 updated_connection_config["realm_id"] = discovered_realm_id
  412:                 conn.execute(
  413:                     """
  414:                     UPDATE e3_erp_connections
  415:                     SET connection_config_json = ?
  416:                     WHERE tenant_id = ?
  417:                       AND workspace_id = ?
  418:                       AND id = ?
  419:                     """,
  420:                     (
  421:                         _db._canonical_json_text(updated_connection_config),
  422:                         tenant,
  423:                         workspace,
  424:                         connection,
  425:                     ),
  426:                 )
  427:         except Exception as exc:  # noqa: BLE001
  428:             conn.execute(
  429:                 "UPDATE e3_erp_oauth_sessions SET session_status = 'FAILED' WHERE id = ?",
  430:                 (str(session["id"]),),
  431:             )
  432:             raise
  433:         conn.execute(
  434:             """
  435:             UPDATE e3_erp_oauth_sessions
  436:             SET session_status = 'CONSUMED',
  437:                 consumed_at = ?
  438:             WHERE id = ?
  439:             """,
  440:             (_db.utc_now_iso(), str(session["id"])),
  441:         )
  442:     erp_repository.set_connection_validation(
  443:         tenant_id=tenant,
  444:         workspace_id=workspace,
  445:         connection_id=connection,
  446:         status="ACTIVE",
  447:     )
  448:     erp_health_service.mark_connection_success(
  449:         tenant_id=tenant,
  450:         workspace_id=workspace,
  451:         connection_id=connection,
  452:     )
  453:     return {
  454:         "connection_id": connection,
  455:         "provider_type": provider,
  456:         "status": "ACTIVE",
  457:         "organization_id": discovered_org_id,
  458:         "organization_stored": bool(discovered_org_id),
  459:         "realm_id": discovered_realm_id,
  460:         "realm_stored": bool(discovered_realm_id),
  461:     }

  503: def refresh_oauth_tokens(
  504:     *,
  505:     tenant_id: str,
  506:     workspace_id: str,
  507:     connection_id: str,
  508: ) -> dict[str, Any]:
  509:     _require_schema()
  510:     tenant, workspace, connection = _normalize_scope(
  511:         tenant_id=tenant_id,
  512:         workspace_id=workspace_id,
  513:         connection_id=connection_id,
  514:     )
  515:     conn_row = erp_repository.get_connection(
  516:         tenant_id=tenant,
  517:         workspace_id=workspace,
  518:         connection_id=connection,
  519:     )
  520:     if conn_row is None:
  521:         raise LookupError("erp_connection_id was not found.")
  522:     provider = _normalize_provider(str(conn_row.get("erp_type") or ""))
  523:     adapter = _adapter_for(provider)
  524:     secret_ref = str(conn_row.get("encrypted_credentials_ref") or "")
  525:     secret_payload = get_connection_secret_payload(
  526:         tenant_id=tenant,
  527:         workspace_id=workspace,
  528:         secret_ref=secret_ref,
  529:     )
  530:     if not secret_payload:
  531:         raise RuntimeError("ERP_OAUTH_TOKEN_MISSING")
  532:     token_payload = _db._normalize_engine3_json_object(
  533:         secret_payload.get("token_payload") or {},
  534:         field_name="token_payload",
  535:     )
  536:     try:
  537:         refreshed = adapter.refresh_oauth_tokens(
  538:             connection_config=conn_row.get("connection_config") or {},
  539:             token_payload=token_payload,
  540:         )
  541:     except Exception as exc:  # noqa: BLE001
  542:         erp_health_service.mark_connection_failure(
  543:             tenant_id=tenant,
  544:             workspace_id=workspace,
  545:             connection_id=connection,
  546:             error_code="ERP_AUTH_REFRESH_FAILED",
  547:             error_message=str(exc),

  640: def get_connection_credentials_for_runtime(
  641:     *,
  642:     tenant_id: str,
  643:     workspace_id: str,
  644:     connection: dict[str, Any],
  645:     auto_refresh: bool = True,
  646: ) -> dict[str, Any]:
  647:     secret_ref = _db._normalize_required_text(
  648:         connection.get("encrypted_credentials_ref"),
  649:         field_name="encrypted_credentials_ref",
  650:     )
  651:     payload = get_connection_secret_payload(
  652:         tenant_id=tenant_id,
  653:         workspace_id=workspace_id,
  654:         secret_ref=secret_ref,
  655:     )
  656:     if payload is None:
  657:         return {}
  658:     token_payload = payload.get("token_payload")
  659:     if isinstance(token_payload, dict):
  660:         normalized = _db._normalize_engine3_json_object(token_payload, field_name="token_payload")
  661:         if not auto_refresh:
  662:             return normalized
  663:         auth_type = str(connection.get("auth_type") or "").strip().upper()
  664:         provider = str(connection.get("erp_type") or "").strip().upper()
  665:         if auth_type == "OAUTH2" and provider in _OAUTH_PROVIDERS and _oauth_token_needs_refresh(normalized):
  666:             refresh_oauth_tokens(
  667:                 tenant_id=tenant_id,
  668:                 workspace_id=workspace_id,
  669:                 connection_id=str(connection.get("id") or ""),
  670:             )
  671:             refreshed_payload = get_connection_secret_payload(
  672:                 tenant_id=tenant_id,
  673:                 workspace_id=workspace_id,
  674:                 secret_ref=secret_ref,
  675:             ) or {}
  676:             refreshed_token_payload = refreshed_payload.get("token_payload")
  677:             if isinstance(refreshed_token_payload, dict):
  678:                 return _db._normalize_engine3_json_object(refreshed_token_payload, field_name="token_payload")
  679:         return normalized
  680:     return _db._normalize_engine3_json_object(payload, field_name="credentials_payload")
